QUICK START
───────────

1. Record IDE Session:
   python ide_recorder_cli.py record --output video.mp4 --duration 300

2. Add AI Narration:
   python ide_recorder_cli.py process --input video.mp4 \
     --add-narration --ai-api-key sk-xxx --quality 1080p

3. Upload to YouTube:
   python ide_recorder_cli.py upload --video video.mp4 \
     --title "My Tutorial" --privacy private

API USAGE EXAMPLE
──────────────────

from src.ide_recorder import *
from pathlib import Path

# Record with event tracking
recorder = ScreenRecorder("video.mp4", fps=30)
events = EventTracker()

recorder.start()
events.start_tracking()

# ... IDE work happens ...

events.track_file_saved(Path("main.py"))
events.track_test_run("test_main", passed=True)

recorder.stop()
events.stop_tracking()

# Detect steps
timeline = TimelineManager()
steps = timeline.detect_logical_steps(events.events)

# Generate narration
narrator = AInarrator(ai_api_key="sk-xxx")
for step in steps:
    text = narrator.generate_narration_for_step(
        step.name, step.description, "Completed successfully"
    )

# Process video
processor = VideoProcessor("video.mp4", "final.mp4")
processor.add_narration(audio)
output = processor.export(quality="1080p")

# Upload
uploader = YouTubeUploader(refresh_token="xxx")
uploader.authenticate()
video_id = uploader.upload_video(output, metadata)
